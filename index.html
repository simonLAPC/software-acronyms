DRY

Don't Repeat Yourself

Every piece of code should have a single, authoritative representation within a system.

  • Change code in one place.

  • Prevents behavior drift.

  • Improves readability.

WET Approach (Repetitive)

# WET: Repetitive file writing logic
import json

def save_user_as_json(user, filename):
    with open(filename, 'w') as f:
        content = json.dumps(user, indent=2)
        f.write(content)
    print(f"Saved {filename}!")

def save_user_as_txt(user, filename):
    with open(filename, 'w') as f:
        content = f"Name: {user['name']}\nEmail: {user['email']}"
        f.write(content)
    print(f"Saved {filename}!")

DRY Approach (Reusable)

# DRY: Abstracted file writing logic
import json

def save_to_file(content, filename):
    with open(filename, 'w') as f:
        f.write(content)
    print(f"Saved {filename}!")

def save_user_as_json(user, filename):
    content = json.dumps(user, indent=2)
    save_to_file(content, filename)

def save_user_as_txt(user, filename):
    content = f"Name: {user['name']}\nEmail: {user['email']}"
    save_to_file(content, filename)
